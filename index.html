1Google Sheets 새 파일 만들기
구글 계정으로 로그인 후 sheets.new 를 브라우저 주소창에 입력하면 바로 생성됩니다.

2Apps Script 편집기 열기
스프레드시트 상단 메뉴에서 확장 프로그램 → Apps Script 클릭
3아래 코드 전체 복사 → 붙여넣기 → 저장(💾)
편집기에 있는 기존 내용을 모두 지우고 아래 코드를 붙여넣으세요.
const SHEET_NAME = "검사결과"; function initSheet(ss) { let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["저장시각","차량번호","검사일","검사자","부대명","항목ID","항목명","결과","비고"]); sheet.getRange(1,1,1,9).setFontWeight("bold").setBackground("#1a3260").setFontColor("white"); } return sheet; } function doPost(e) { try { const ss=SpreadsheetApp.getActiveSpreadsheet(); const sheet=initSheet(ss); const data=JSON.parse(e.postData.contents); const ts=new Date().toLocaleString("ko-KR"); const rows=sheet.getDataRange().getValues(); for(let i=rows.length-1;i>=1;i--){ if(rows[i][1]===data.vehicleNo&&rows[i][5]===data.itemId) sheet.deleteRow(i+1); } sheet.appendRow([ts,data.vehicleNo,data.inspDate,data.inspector,data.unitName,data.itemId,data.itemName,data.result,data.note||""]); return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:"error",msg:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const ss=SpreadsheetApp.getActiveSpreadsheet(); const sheet=ss.getSheetByName(SHEET_NAME); let jsonData; if(!sheet){ jsonData=JSON.stringify({status:"ok",rows:[]}); } else { const rows=sheet.getDataRange().getValues(); const headers=rows[0]; const data=rows.slice(1).map(r=>{let obj={};headers.forEach((h,i)=>obj[h]=r[i]);return obj;}); jsonData=JSON.stringify({status:"ok",rows:data}); } const cb=e.parameter&&e.parameter.callback; if(cb) return ContentService.createTextOutput(cb+"("+jsonData+");").setMimeType(ContentService.MimeType.JAVASCRIPT); return ContentService.createTextOutput(jsonData).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:"error",msg:err.toString()})).setMimeType(ContentService.MimeType.JSON); } }
4웹 앱으로 배포하기
Apps Script 편집기 오른쪽 상단 배포 → 새 배포 클릭
① 유형 선택 → 웹 앱
② 다음 사용자로 실행 → 나(내 Google 계정)
③ 액세스 권한 → 모든 사용자
④ 배포 버튼 클릭 → 권한 승인
⑤ 웹 앱 URL 복사
💡 URL은 https://script.google.com/macros/s/...../exec 형태입니다.
